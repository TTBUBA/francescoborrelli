🎮 Come Creare il Movimento del Player

Pubblicato il 7 Giugno 2026 | Lettura: 5 minuti

Introduzione

In questo tutorial vedremo come creare un sistema di movimento fluido e reattivo per il tuo player in Unity. Useremo input da tastiera e applicheremo fisica basilare.

Step 1: Setup del Player

Innanzitutto, crea un cubo o importa un modello 3D. Aggiungi un Rigidbody per la fisica:


using UnityEngine;

public class PlayerMovement : MonoBehaviour
{
    public float moveSpeed = 5f;
    private Rigidbody rb;
    private Vector3 moveDirection;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
    }

    void Update()
    {
        HandleInput();
    }

    void FixedUpdate()
    {
        Move();
    }

    void HandleInput()
    {
        float x = Input.GetAxis("Horizontal");
        float z = Input.GetAxis("Vertical");
        moveDirection = new Vector3(x, 0, z).normalized;
    }

    void Move()
    {
        rb.velocity = new Vector3(moveDirection.x * moveSpeed, rb.velocity.y, moveDirection.z * moveSpeed);
    }
}

Step 2: Aggiungi Rotazione

Per far ruotare il player nella direzione del movimento:


void HandleInput()
{
    float x = Input.GetAxis("Horizontal");
    float z = Input.GetAxis("Vertical");
    moveDirection = new Vector3(x, 0, z).normalized;

    if (moveDirection != Vector3.zero)
    {
        Quaternion newRotation = Quaternion.LookRotation(moveDirection);
        transform.rotation = Quaternion.Slerp(transform.rotation, newRotation, Time.deltaTime * 10f);
    }
}

Step 3: Aggiungi Salto

Adesso aggiungiamo il salto al nostro player:


public float jumpForce = 5f;
public float groundDrag = 5f;
private bool isGrounded;
private float groundDistance = 0.4f;
public LayerMask groundLayer;

void Update()
{
    HandleInput();
    CheckGround();
}

void CheckGround()
{
    isGrounded = Physics.CheckSphere(transform.position - Vector3.up * 0.5f, groundDistance, groundLayer);
    rb.drag = isGrounded ? groundDrag : 0;
}

void HandleInput()
{
    float x = Input.GetAxis("Horizontal");
    float z = Input.GetAxis("Vertical");
    moveDirection = new Vector3(x, 0, z).normalized;

    if (Input.GetKeyDown(KeyCode.Space) && isGrounded)
    {
        Jump();
    }
}

void Jump()
{
    rb.velocity = new Vector3(rb.velocity.x, 0, rb.velocity.z);
    rb.AddForce(Vector3.up * jumpForce, ForceMode.Impulse);
}

Conclusione

Perfetto! Ora hai un sistema di movimento base con salto. Puoi espanderlo aggiungendo corsa, doppio salto, e animazioni! 🚀

Suggerimenti:

  • Regola moveSpeed e jumpForce per trovare l'equilibrio perfetto
  • Aggiungi una particella al salto per effetti visivi
  • Crea animazioni sincronizzate con il movimento
